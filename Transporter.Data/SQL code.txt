CREATE TABLE PAKAGE
(
PAKAGE_ID NUMERIC PRIMARY KEY,
PSIZE VARCHAR(6) CHECK(PSIZE = 'SMALL' OR PSIZE = 'MEDIUM' OR PSIZE = 'LARGE'),

PRECEIVER_ID NUMERIC REFERENCES dbo.CUSTOMER(CUSTOMER_ID),
PDRIVER_ID NUMERIC REFERENCES dbo.DRIVER(DRIVER_ID)
